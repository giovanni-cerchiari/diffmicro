60 {

64  FLOAT min,max;
65  FLOAT most_similar;
66  std::vector<FLOAT> euristic;
67  FLOAT *row;
68 
69  UINT windowborder = (windowside-1)/2;
70  euristic.resize(windowside*windowside);
71 
72 
73  xbegin = windowborder + 1; xend = dimx-(windowborder + 1);
74  ybegin = windowborder + 1; yend = dimy-(windowborder + 1);
75 
76  dim = dimx*dimy;
77  most_similar = mat[ybegin*dimx+xbegin];
78  similarity = fabs(mat[ybegin*dimx+xbegin]-val);
79  i=xbegin; j=ybegin;

81  {
82  row = &(mat[y*dimx]);
83  for(x=xbegin; x<xend; ++x)
84  {
85  if(similarity > fabs(row[x]-val))
86  {
87  similarity = fabs(row[x]-val);
88  most_similar = row[x];
89  i = x; j = y;
90  }
91  }
92  }
93 
94  i -= windowborder;
95  j -= windowborder;
96 
97  for(y=0; y<windowside; ++y)
98  for(x=0; x<windowside; ++x)
99  euristic[y*windowside+x]=mat[(y+j)*dimx+(x+i)];
100 
101  sort(euristic.begin(), euristic.end());
102 
103  max = euristic[0];
104  min = euristic[0];
105  for(i=1; i<2*windowside; ++i)
106  {
107  if(euristic[i]>max)max = euristic[i];
108  if(euristic[i]<min)min = euristic[i];
109  }
110 
111  if(fabs(max-val)>fabs(val-min)) similarity = fabs(max-val);
112  else similarity = fabs(min-val);
