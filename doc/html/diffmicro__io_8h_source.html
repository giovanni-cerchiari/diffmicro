393 {
394  unsigned short size;
395  FILE *pfile;
396 
397  pfile = fopen(filename.c_str(), "wb");
398  if(NULL == pfile)
399  {
400  std::cerr <<"error attempting to write : "<<filename<<std::endl;
401  return false;
402  }
403 
404  size = sizeof(FLOAT);
405  fwrite(&size, sizeof(unsigned short), 1, pfile);
406 
407  size = (unsigned short)(dim);
408  fwrite(&size, sizeof(unsigned short), 1, pfile);
409 
410  fwrite(vet, sizeof(FLOAT), dim, pfile);
411 
412  fclose(pfile);
413 
414  return true;

514 {
515  std::fstream fin;
