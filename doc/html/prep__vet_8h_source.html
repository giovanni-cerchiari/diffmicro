126 
127  val = new FLOAT[dimvet];
128  freq = new INT[dimvet];
129

134  dim_freq = 0;
135  for(i=0; i<dimvet; ++i)

137  new_val = *(vet + i);
138  flg_found = false;
139  for(j=0; j<dim_freq; ++j)
140  {
141  if(fabs(val[j] - new_val) < (0.0000000001 * val[j]))
142  {
143  ++freq[j];
144  flg_found = true;
145  break;
146  }
147  }
148  if(flg_found == false)
149  {
150  val[dim_freq] = new_val;
151  ++freq[dim_freq];
152  ++dim_freq;
153  }
154  }
155  maxminary(dim_freq, freq, max, pos_max, min, pos_min);
156 
157  vall = val[pos_max];
158  delete[] val;
159  delete[] freq;
160 
161  return (vall);
162 }
