219 void complex_to_module_phase(INDEX dim, CUFFT_COMPLEX *in, TYPEFLOAT *outm, TYPEFLOAT *outp)
220 {
221  INDEX i;
222  for (i = 0; i < s_load_image.dim; ++i)
223  {
224  outm[i] = sqrt(in[i].x* in[i].x + in[i].y* in[i].y);
225  outp[i] = atan2(in[i].y, in[i].x);
226  }
227 
228 }
