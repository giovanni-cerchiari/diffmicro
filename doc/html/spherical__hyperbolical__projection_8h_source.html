48 template<typename FLOAT, typename UINT>
49 void spheric_hyperbolic_indices_for_projection(FLOAT r, UINT dimx, UINT dimy, FLOAT scale[], FLOAT *indout, int type = 0)
50 {
51  UINT i, j;
52  FLOAT *indoutrow;
53  FLOAT x, y;
54  FLOAT theta;
55 
56  FLOAT(*_sin)(FLOAT);
57  FLOAT(*_cos)(FLOAT);
58  FLOAT(*_tan)(FLOAT);
59  FLOAT(*_atan)(FLOAT);
60  FLOAT(*_asin)(FLOAT);
61 
62  switch (type)
63  {
64  case HYPERBOLICAL_PROJECTION:
65  _sin = sinh;
66  _cos = cosh;
67  _tan = tanh;
68  _asin = asinh;
69  _atan = atanh;
70  break;
71  case SPHERICAL_PROJECTION:
72  _sin = sin;
73  _cos = cos;
74  _tan = tan;
75  _asin = asin;
76  _atan = atan;
77 
78  default:
79  break;
80  }
81 
82  for (j = 0; j < dimy; ++j)
83  {
84  indoutrow = &(indout[2 * j * dimx]);
85  for (i = 0; i < dimx; ++i)
86  {
87  y = ((FLOAT)(j)-0.5*(FLOAT)(dimy-1)) * scale[1];
88  x = ((FLOAT)(i)-0.5*(FLOAT)(dimx-1)) * scale[0];
89 
90  theta = _atan(sqrt(x*x + y*y) / r);
91 
92 // std::cerr << "cos(theta = " << theta <<") = "<<_cos(theta)<< std::endl;
93 
94  indoutrow[2 * i ] = x*_cos(theta);
95  indoutrow[2 * i + 1] = y*_cos(theta);
96 // std::cerr << "(" << i << "," << j << ") \t (" << x << "," << y << ") \t (" << x*_cos(theta) << "," << y*_cos(theta) << ")" << std::endl;
97  }
98  }

100 }
